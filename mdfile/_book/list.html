4.1 what would happen if you combine vectors of different class

vec <- c(1,2,3)
char_vec <- c("R", "Rstudio", "Github")
logic_vec <- c(TRUE, FALSE, TRUE, FALSE)
out_list <- c(vec, char_vec, logic_vec)
out_list
##  [1] "1"       "2"       "3"       "R"       "Rstudio" "Github"  "TRUE"   
##  [8] "FALSE"   "TRUE"    "FALSE"
class(out_list)
## [1] "character"

4.11 matrix example with list

Seinfeld <- c(13, 11)
Friends <- c(53, 49)
Office <- c(40, 35)

pop_index <- c(Seinfeld, Friends, Office)

tv_matrix <- matrix(pop_index, byrow = TRUE, nrow = 3,
                    dimnames = list(c("Seinfeld", "Friends", "Office"),
                                    c("year_2018", "year_2019")))

tv_matrix
##          year_2018 year_2019
## Seinfeld        13        11
## Friends         53        49
## Office          40        35
